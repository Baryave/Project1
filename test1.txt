#include <iostream>//подключает стандартный библиотечный заголовочный файл для работы с потоками ввода-вывода
#include <stdio.h>//подключение библиотеки ввода и вывода

int main()
{/*предусл: вводится сопротивление r1 и сопротивление r2 с консоли
 постусл: производится вычисление сопротивления соединения сначала при параллельном соединении сопротивлений,
 а затем при последовательном соединении сопротивлений*/
	setlocale(LC_ALL, "Russian");//переключает раскладку на русский язык
	double r1, r2, parallelno, posledovatelno;//задаем переменные, коториые являются вещественным числом
	printf("Введите r1: ");//выводит на экран заданное в скобках
	scanf("%lf", &r1);//ввод значения переменной r1
	printf("Введите r2: ");//выводит на экран заданное в скобках
	scanf("%lf", &r2);//ввод значения переменной r2
	parallelno = (r1 * r2) / (r1 + r2);// формула вычисления сопротивления при параллельном соединении сопротивлений
	printf("Паралельно: %lf\n", parallelno);//вывод результата формулы параллельного соединения
	posledovatelno = r1 + r2;// формула вычисления сопротивления при последовательном соединении сопротивлений
	printf("Последовательно: %lf\n", posledovatelno);//вывод результата формулы последовательного соединения
	return 0;//возвращаемое значение функции
}
